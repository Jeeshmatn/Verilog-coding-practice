//  D Flip-Flop (Positive Edge-Triggered)

// Write a Verilog module for a positive edge-triggered D Flip-Flop with asynchronous reset and clock input.

// Inputs: clk, reset, D                     Output: Q

// On positive edge of clk, Q should take value of D

// If reset is high, Q should be 0 regardless of the clocke your design here










module d_ff(
  input clk,reset,d,
  output reg q
);
  
  always @(posedge clk or posedge reset)begin
    
    if(reset)
      q<=0;
    else 
      q<=d;
    
  end
  
endmodule
